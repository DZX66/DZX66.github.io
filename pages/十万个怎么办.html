pyinstaller的使用

pyinstaller是一个将Python代码打包成可执行文件的工具，使用方法如下：


pyinstaller -F xxx.py -w -i xxx.ico

其中，-F表示打包成单个可执行文件，-w表示不显示控制台，-i表示设置图标。

在未登录的情况下，b站干掉登录窗口

可以实现，但无法恢复全屏，因为浏览器安全策略不允许脚本自发全屏

代码如下：


// ==UserScript==
// @name         B站（bilibili）自动续播因未登录而暂停的视频 (Bilibili: Continue playing without logging-in)
// @namespace    http://tampermonkey.net/
// @version      0.3
// @description  解决B站（bilibili）因未登录而自动暂停视频并弹出登录窗口的问题 / Solve the problem of Bilibili automatically pausing video and popping up a login window because it is not logged in
// @author       TheBeacon&dzxhbwls
// @match        https://www.bilibili.com/video/*
// @icon         data:image/gif;base64,R0lGODlhAQABAAAAACH5BAEKAAEALAAAAAABAAEAAAICTAEAOw==
// @grant        none
// @license      MIT
// ==/UserScript==


(function () {
  'use strict';
  var is_fullscreen = 0;
  var config = { attributes: true, attributeFilter: ["class"] };

  function monitorHTMLCollection(collection, callback) {
    let lengthBefore = collection.length;

    // 定时器检查集合是否有变化
    setInterval(function () {
      if (collection.length !== lengthBefore) {
        // 如果集合长度发生变化，执行回调函数
        callback();
        lengthBefore = collection.length; // 更新长度记录
      }
    }, 100); // 间隔时间可以根据需要调整
  }
  var fs = document.getElementsByClassName("bpx-player-ctrl-btn bpx-player-ctrl-full");

  window.onload = function () {

    monitorHTMLCollection(fs, function () {
      fs = fs[0];
      fs.addEventListener("click", () => {
        is_fullscreen = 1 - is_fullscreen;
        if (is_fullscreen) { console.log("fullscreened"); } else { console.log("unfullscreened"); }
      });
    });

  };

  var target = document.getElementsByClassName("bpx-player-row-dm-wrap")[0];
  var observer = new MutationObserver(function (motationList, observer) {
    setTimeout(function () {
      if (document.getElementsByClassName("bili-mini-mask").length > 0) {
        document.getElementsByClassName("bili-mini-close-icon")[0].click();
        if (!is_fullscreen) {
            // 由于浏览器安全策略，无法自动全屏
          document.getElementsByClassName("bpx-player-ctrl-btn bpx-player-ctrl-play")[0].click();
        }
        observer.takeRecords();
      }
    }, 1000);
  });
  observer.observe(target, config);
})();
